class Config:
    def __init__(self):
        self._config = self._load_config()
        self._validate()

    def _load_config(self) -> Dict[str, Any]:
        config = DEFAULT_CONFIG.copy()

        if not HAS_YAML:
            print("[WARN] PyYAML not installed, using default config")
            print("[INFO] Install with: pip install pyyaml")
            return config

        # Load from file if exists
        config_path = None
        if LOCAL_CONFIG_PATH.exists():
            config_path = LOCAL_CONFIG_PATH
        elif USER_CONFIG_PATH.exists():
            config_path = USER_CONFIG_PATH

        if config_path:
            with open(config_path) as f:
                user_config = yaml.safe_load(f)
                if user_config:
                    self._deep_merge(config, user_config)

        return config

    def _deep_merge(self, base: Dict, override: Dict):
        for key, value in override.items():
            if key in base and isinstance(base[key], dict) and isinstance(value, dict):
                self._deep_merge(base[key], value)
            else:
                base[key] = value

    def _validate(self):
        # Expand data directory path
        data_dir = Path(self._config["data_dir"]).expanduser()
        self._config["data_dir"] = str(data_dir)

        self.ocr_dir = data_dir / "ocr"
        self.embedding_dir = data_dir / "embeddings"
        self.db_path = data_dir / "database.db"

            d.mkdir(parents=True, exist_ok=True)

        self._validate_embedding_config()

    def _validate_embedding_config(self):
        """Validate embedding configuration based on mode."""
        if not self.is_worker_enabled("embedding"):
            return

        mode = self.get("workers.embedding.mode", "text-image-hybrid")
        valid_modes = ["text-only", "image-only", "text-image-hybrid"]

        if mode not in valid_modes:
            raise ValueError(
                f"Invalid embedding mode: '{mode}'. "
                f"Must be one of: {', '.join(valid_modes)}"
            )

        # Validate based on mode
        errors = []

        if mode in ("text-only", "text-image-hybrid"):
            text_config = self.get("workers.embedding.text", {})
            if not text_config.get("base_url"):
                errors.append("workers.embedding.text.base_url is required")
            if not text_config.get("model"):
                errors.append("workers.embedding.text.model is required")

        if mode in ("image-only", "text-image-hybrid"):
            image_config = self.get("workers.embedding.image", {})
            if not image_config.get("base_url"):
                errors.append("workers.embedding.image.base_url is required")
            if not image_config.get("model"):
                errors.append("workers.embedding.image.model is required")

        if errors:
            raise ValueError(
                "Embedding configuration errors:\n  - " + "\n  - ".join(errors)
            )

class DependencyChecker:
    # Maps feature name to (binary_name, ubuntu_package, check_args)
    BINARY_DEPS = {
        "tesseract": ("tesseract", "tesseract-ocr", ["--version"]),
    }

    # Maps language code to tesseract language package
    TESSERACT_LANGS = {
        "eng": "tesseract-ocr-eng",
        "chi_sim": "tesseract-ocr-chi-sim",
        "chi_tra": "tesseract-ocr-chi-tra",
        "fra": "tesseract-ocr-fra",
        "deu": "tesseract-ocr-deu",
        "spa": "tesseract-ocr-spa",
        "jpn": "tesseract-ocr-jpn",
        "kor": "tesseract-ocr-kor",
        "rus": "tesseract-ocr-rus",
    }

    @classmethod
    def check_binary(cls, name: str, raise_error: bool = True) -> bool:
        binary_name, ubuntu_pkg, check_args = cls.BINARY_DEPS[name]

        if shutil.which(binary_name):
            return True

        if raise_error:
            raise DependencyError(
                f"'{binary_name}' is required but not found in PATH",
                f"sudo apt-get install {ubuntu_pkg}"
            )
        return False

    @classmethod
    def check_tesseract_langs(cls, languages: List[str]) -> Tuple[List[str], List[str]]:
        available = []
        missing = []

        try:
            # Get installed languages
            result = subprocess.run(
                ["tesseract", "--list-langs"],
                capture_output=True,
                text=True,
                timeout=5
            )
            installed = set(result.stdout.strip().split('\n')[1:])
        except:
            return [], languages

        for lang in languages:
            if lang in installed or lang == "eng" and "osd" in installed:
                available.append(lang)
            else:
                missing.append(lang)

        return available, missing

    @classmethod
    def get_lang_install_cmd(cls, languages: List[str]) -> str:
        packages = [cls.TESSERACT_LANGS.get(lang, f"tesseract-ocr-{lang}")
                    for lang in languages]
        return f"sudo apt-get install {' '.join(packages)}"

    @classmethod
    def check_all(cls, skip_optional: bool = False):
        errors = []

        # Check tesseract if OCR enabled
        if config.is_worker_enabled("ocr") and not skip_optional:
            try:
                cls.check_binary("tesseract")
                # Check configured languages

                available, missing = cls.check_tesseract_langs(langs)
                if missing:
                    install_cmd = cls.get_lang_install_cmd(missing)
                    print(f"[WARN] Missing OCR languages: {missing}")
                    print(f"[INFO] Install with: {install_cmd}")
                    # Continue with available languages
                    print(f"[INFO] Continuing with available languages: {available}")
            except DependencyError as e:
                errors.append(e)

        if errors:
            for e in errors:
                print(e, file=sys.stderr)
            sys.exit(1)

        with sqlite3.connect(self.db_path) as conn:
            conn.execute("""
                CREATE TABLE IF NOT EXISTS screenshots (
                    id INTEGER PRIMARY KEY AUTOINCREMENT,
                    timestamp REAL NOT NULL,
                    screenshot_path TEXT NOT NULL,
                    ocr_path TEXT,
                    embedding_path TEXT,
                    window_title TEXT,
                    ocr_text TEXT,
                    created_at REAL DEFAULT (strftime('%s', 'now')),

                CREATE INDEX IF NOT EXISTS idx_ocr_text ON screenshots(ocr_text)
            """)

    def insert_screenshot(self, timestamp: float, screenshot_path: str) -> int:
        with sqlite3.connect(self.db_path) as conn:
            cursor = conn.execute(
                "INSERT OR IGNORE INTO screenshots (timestamp, screenshot_path) VALUES (?, ?)",
                (timestamp, screenshot_path)
            )
            conn.commit()

    def update_ocr(self, timestamp: float, ocr_path: str, ocr_text: str):
        with sqlite3.connect(self.db_path) as conn:
            conn.execute(

        with sqlite3.connect(self.db_path) as conn:

class BM25Search:
    def __init__(self):
        self.db = Database(DB_PATH)
        self.k1 = BM25_K1
        self.b = BM25_B
        self.doc_lengths = {}
        self.avg_dl = 0
        self.doc_freqs = defaultdict(int)
        self.inverted_index = defaultdict(list)
        self.N = 0
        self._build_index()

    def _tokenize(self, text: str) -> List[str]:
        if NLTK_AVAILABLE:
            return word_tokenize(text.lower())
        return re.findall(r'\b\w+\b', text.lower())

    def _build_index(self):
        records = self.db.get_all_ocr_text()
        total_length = 0

        for doc_id, text in records:
            if not text:
                continue

            tokens = self._tokenize(text)
            self.doc_lengths[doc_id] = len(tokens)
            total_length += len(tokens)

            # Build inverted index
            term_counts = defaultdict(int)
            for token in tokens:
                term_counts[token] += 1

            for term, freq in term_counts.items():
                self.inverted_index[term].append((doc_id, freq))
                self.doc_freqs[term] += 1

        self.N = len(self.doc_lengths)
        self.avg_dl = total_length / self.N if self.N > 0 else 0

    def search(self, query: str, top_k: int = 20) -> List[Tuple[int, float]]:
        query_terms = self._tokenize(query)
        scores = defaultdict(float)

        for term in query_terms:
            if term not in self.inverted_index:
                continue

            df = self.doc_freqs[term]
            idf = math.log((self.N - df + 0.5) / (df + 0.5) + 1)

            for doc_id, tf in self.inverted_index[term]:
                dl = self.doc_lengths[doc_id]
                denom = self.k1 * (1 - self.b + self.b * (dl / self.avg_dl)) + tf
                score = idf * (tf * (self.k1 + 1)) / denom
                scores[doc_id] += score

        results = sorted(scores.items(), key=lambda x: x[1], reverse=True)
        return results[:top_k]

class TextEmbeddingSearch:
    """Search using text embeddings (OCR content)."""

    def __init__(self, config: Config = None, db: Database = None):
        self.config = config or Config()
        self.db = db or Database(self.config.db_path)
        self.client = EmbeddingClient(
            self.config.get_text_embedding_config(),
            "text"
        )
        self.dimension = self.config.get("workers.embedding.text.dimension")

        if not self.dimension:
            raise RuntimeError(
                "Text embedding dimension not configured. "
                "Run 'flashback config test-embedding --type text' to detect dimension."
            )

    def search(self, query: str, top_k: int = 20) -> List[Tuple[int, float]]:
        query_embedding = self.client.get_text_embedding(query)
        return self._search_by_vector(query_embedding, top_k, ".text.npy")

    def _search_by_vector(self, query_vec: np.ndarray, top_k: int, suffix: str) -> List[Tuple[int, float]]:
        records = self.db.get_unprocessed_embeddings(limit=10000)
        scores = []
        query_norm = np.linalg.norm(query_vec)

        for record in records:
            if not record.embedding_path:
                continue

            try:
                emb_path = Path(record.embedding_path)
                # For hybrid mode, look for .text.npy file
                if suffix:
                    emb_path = emb_path.with_suffix(suffix)

                if not emb_path.exists():
                    continue

                emb = np.load(emb_path)
                if len(emb) != self.dimension:
                    continue

                similarity = np.dot(query_vec, emb) / (query_norm * np.linalg.norm(emb))
                scores.append((record.id, float(similarity)))
            except Exception as e:
                continue

        scores.sort(key=lambda x: x[1], reverse=True)
        return scores[:top_k]

@app.get("/api/search")
async def search(
    q: str = Query(..., description="Search query"),
    mode: str = Query("bm25", description="bm25 or embedding"),
    start_date: Optional[str] = None,
    end_date: Optional[str] = None,
    limit: int = 20
):
    from config import config

    # Check if search method is enabled
    if mode == "embedding" and not config.is_search_enabled("embedding"):
        return {"error": "Embedding search is not enabled"}, 400
    if mode == "bm25" and not config.is_search_enabled("bm25"):
        return {"error": "BM25 search is not enabled"}, 400

    # Time range filtering
    start_ts = None
    end_ts = None
    if start_date:
        start_ts = datetime.fromisoformat(start_date).timestamp()
    if end_date:
        end_ts = datetime.fromisoformat(end_date).timestamp()

    # Perform search
    if mode == "embedding":
        if embedding_search is None:
            return {"error": "Embedding search not available - install dependencies"}, 503
        results = embedding_search.search_by_text(q, top_k=limit)
    else:
        if bm25_search is None:
            return {"error": "BM25 search not available"}, 503
        results = bm25_search.search(q, top_k=limit)

    # Fetch full records
    records = []
    for doc_id, score in results:
        with sqlite3.connect(DB_PATH) as conn:
            conn.row_factory = sqlite3.Row
            row = conn.execute(
                "SELECT * FROM screenshots WHERE id = ?", (doc_id,)
            ).fetchone()

        if row:
            record = dict(row)
            # Filter by time range if specified
            if start_ts and record['timestamp'] < start_ts:
                continue
            if end_ts and record['timestamp'] > end_ts:
                continue

            record['score'] = score
            record['timestamp_formatted'] = datetime.fromtimestamp(record['timestamp']).isoformat()
            records.append(record)

    return {"results": records, "query": q, "mode": mode}

@app.get("/api/neighbors/{timestamp}")

sudo apt-get install tesseract-ocr

# Optional: install language packs for OCR
sudo apt-get install tesseract-ocr-chi-sim  # Chinese Simplified
sudo apt-get install tesseract-ocr-chi-tra  # Chinese Traditional
sudo apt-get install tesseract-ocr-jpn      # Japanese
# (see SECTION 17 for full list)

# 5. Create config file (optional - uses defaults if missing)
mkdir -p ~/.config/screenshot-searcher
cp config.example.yaml ~/.config/screenshot-searcher/config.yaml
# Edit to enable/disable workers and set preferences

# 6. Configure embedding API (optional, for semantic search)
# Example with Ollama (local):
#   ollama pull llava
#   flashback config set workers.embedding.api.base_url "http://localhost:11434/v1"
#   flashback config set workers.embedding.api.model "llava"
#   flashback config test-embedding --write  # Detects dimension and writes to config

# Example with OpenAI:
#   export OPENAI_API_KEY="sk-..."
#   flashback config set workers.embedding.api.base_url "https://api.openai.com/v1"
#   flashback config set workers.embedding.api.model "text-embedding-3-small"
#   flashback config set workers.embedding.api.api_key "${OPENAI_API_KEY}"
#   flashback config test-embedding --write

# 7. Run the workers (captures screenshots and processes)
flashback serve --daemon

# 8. Start the web interface (in a new terminal)
flashback webui

# 9. Open browser to http://localhost:8080
"""

"""
ASSUMPTIONS (1920x1080 display):
- PNG screenshot: ~200 KB (with compression)
- OCR text file: ~5 KB average
- Embedding: ~2-12 KB (varies by API: 768-dim ~3 KB, 4096-dim ~16 KB)

PER MINUTE: ~210-220 KB
PER HOUR: ~12-13 MB
PER DAY: ~300-320 MB
PER WEEK: ~2.1-2.2 GB
PER MONTH: ~9-10 GB

With 7-day retention: ~2-3 GB total storage
(Depends on embedding API dimension)

To reduce storage:
- Lower screenshot quality (config.py)
- Reduce frequency (e.g., every 5 minutes)
- Shorter retention period
- Store only thumbnails for search results
"""

  SS_DATA_DIR=/path/to/data         # Override data directory

TROUBLESHOOTING:

1. "tesseract not found" error:
   sudo apt-get install tesseract-ocr

2. "Missing OCR language" warning:
   Check available: tesseract --list-langs
   Install more:   sudo apt-get install tesseract-ocr-<lang>

3. "pytesseract not found" error:
   pip install pytesseract
   OR enable it: pip install -e ".[ocr]"

4. "Embedding API not configured" error:
   Set workers.embedding.api.base_url and model in config.yaml
   Example for Ollama: base_url: "http://localhost:11434/v1"
   Example for OpenAI: base_url: "https://api.openai.com/v1"

5. "Embedding dimension not configured" error:
   Run: flashback config test-embedding --write
   This tests the API and automatically sets the dimension

6. Embedding dimension mismatch warning:
   The stored embeddings don't match the configured dimension.
   This happens when changing embedding models.
   Fix: flashback purge-embeddings --yes
   Then restart the workers to regenerate embeddings.

7. High memory usage:
   Reduce batch sizes in config.yaml:
     workers.ocr.batch_size: 2
     workers.embedding.batch_size: 1

"""
USAGE:

  flashback search "TODO" -F json --from 2024-03-01 -t 2024-03-15
  flashback search "invoice" --preview -n 10

  flashback search --image screenshot.png
  flashback search --image photo.jpg --search-mode image_embedding_only

  flashback search "error message" --image screenshot.png

"""

   - Window title

   - OCR text excerpt (first 100-150 chars, truncated with ...)
4. User presses number (1-9) or 'v' to view

VIEWING OPTIONS:

Option A: Text excerpt (default in search results)
  Pros: Fast, terminal-native, searchable, copy-paste friendly
  Cons: No visual context, formatting lost
  Use: Initial triage, quick identification

Option B: System image viewer (on selection)
  Pros: Native zoom, pan, fullscreen, fast GPU rendering
  Cons: Requires X/Wayland, different app window
  Use: Detailed inspection, reading small text, visual context

Option C: Web UI
  Pros: Integrated view, neighbors timeline, zoom controls
  Cons: Requires browser, more resource usage
  Use: Deep research, browsing history

RECOMMENDED CLI WORKFLOW:
  $ flashback search "deploy failed"
  # Shows 10 results with text excerpts
  # Press 3 to open #3 in image viewer
  # Image viewer opens (feh, eog, xdg-open)
  # Press 'n' in viewer for next result (if supported)
  # Close viewer, back to CLI

CONFIGURABLE IMAGE VIEWER:
  In config.yaml:
    viewer:
      command: "feh"           # or "eog", "xdg-open", "gimp"
      args: ["--scale-down", "{path}"]
      neighbor_args: ["--start-at", "{path}", "{neighbor_paths}"]

  Default behavior:
    Linux:   xdg-open {path}
    macOS:   open {path}
    Windows: start {path}

WHY SYSTEM VIEWER OVER BUILT-IN:
1. No reinventing the wheel - image viewers are mature
2. User preference - respects their chosen tool
3. Feature parity - get zoom, rotate, metadata for free
4. SSH friendly - works with X forwarding
5. Low resource - no embedded web view or image lib dependencies

EDGE CASES:
- SSH without X11: fall back to ASCII art or text only
- Terminal doesn't support images: use text excerpts only
- Image viewer not found: error with install suggestions
